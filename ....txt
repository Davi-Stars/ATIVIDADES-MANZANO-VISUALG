início
  // Exerciço 3.10-C: Eleição treta! Vamos contar os votos
  var votosA, votosB, votosC, nulos, brancos, total: inteiro
  var percVotos, percA, percB, percC, percNulos, percBrancos: real
  leia(votosA, votosB, votosC, nulos, brancos)
  total <- votosA + votosB + votosC + nulos + brancos
  percVotos <- (votosA + votosB + votosC) / total * 100
  percA <- votosA / total * 100
  percB <- votosB / total * 100
  percC <- votosC / total * 100
  percNulos <- nulos / total * 100
  percBrancos <- brancos / total * 100
  escreva("Total de eleitores: ", total)
  escreva("Percentual de votos válidos: ", percVotos)
  escreva("Percentual de votos A: ", percA)
  escreva("Percentual de votos B: ", percB)
  escreva("Percentual de votos C: ", percC)
  escreva("Percentual de votos nulos: ", percNulos)
  escreva("Percentual de votos brancos: ", percBrancos)
fim
